\begin{eqnarray*}
    \bar{x}_n &=& \sum_{i = 1}^{n - 1} \frac{x_i}{n} + \frac{x_n}{n} \\
    &=& \frac{n - 1}{n} \bar{x}_{n - 1} + \frac{x_n}{n} \\
    &=& \bar{x}_{n - 1} + \frac{x_n - \bar{x}_{n - 1}}{n} \\
\end{eqnarray*}
